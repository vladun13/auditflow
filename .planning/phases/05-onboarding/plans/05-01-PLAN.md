### Task 05-01-01: Create useOnboarding hook + test

**File:** `src/hooks/useOnboarding.ts` (new)
**Test:** `src/hooks/useOnboarding.test.ts` (new)

```typescript
// src/hooks/useOnboarding.ts
import { useState, useEffect } from 'react'

const ONBOARDING_KEY = 'auditflow_onboarding_done'
const TUTORIAL_KEY = 'auditflow_tutorial_done'
const PROFILE_KEY = 'auditflow_onboarding_profile'

export function useOnboarding() {
  const [showWelcome, setShowWelcome] = useState(false)

  useEffect(() => {
    const done = localStorage.getItem(ONBOARDING_KEY)
    if (!done) setShowWelcome(true)
  }, [])

  const completeOnboarding = () => {
    localStorage.setItem(ONBOARDING_KEY, 'true')
    setShowWelcome(false)
  }

  const completeTutorial = () => {
    localStorage.setItem(TUTORIAL_KEY, 'true')
  }

  const saveProfile = (profile: Record<string, string>) => {
    localStorage.setItem(PROFILE_KEY, JSON.stringify(profile))
  }

  return { showWelcome, completeOnboarding, completeTutorial, saveProfile }
}
```

**Tests to write:**
- renders without showing welcome when localStorage flag is set
- shows welcome when localStorage flag is absent
- completeOnboarding sets flag and hides modal
- saveProfile serializes to localStorage

**Verify:** `/opt/homebrew/bin/node node_modules/.bin/vitest run src/hooks/useOnboarding.test.ts`
